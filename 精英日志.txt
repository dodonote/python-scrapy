import re
pattern = re.compile('[a-zA-Z]')
result = pattern.findall('as3SiOPdj#@23awe')
print result
# ['a', 's', 'S', 'i', 'O', 'P', 'd', 'j', 'a', 'w', 'e']

        platform.platform()   #获取操作系统名称及版本号，'Windows-7-6.1.7601-SP1'
